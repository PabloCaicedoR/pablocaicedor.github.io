import numpy as np
import cv2


def generate_striped_image(N, M, K, orientation="vertical"):
    """
    Genera una imagen de N×M pixeles con bandas alternas (blanco/negro)
    de ancho K, en orientación vertical u horizontal.

    Parámetros
    ----------
    N : int
        Altura de la imagen en pixeles.
    M : int
        Anchura de la imagen en pixeles.
    K : int
        Ancho de cada banda en pixeles.
    orientation : {'vertical', 'horizontal'}
        Orientación de las bandas. 'vertical' crea franjas verticales;
        'horizontal' crea franjas horizontales.

    Devuelve
    -------
    img : np.ndarray
        Imagen en escala de grises (dtype uint8) con valores 0 o 255.
    """
    # Crear imagen en negro
    img = np.zeros((N, M), dtype=np.uint8)

    # Alternar franjas
    if orientation == "vertical":
        for start in range(0, M, K):
            # Determinar color de la franja (0 o 255)
            color = 255 if ((start // K) % 2) == 0 else 0
            img[:, start : start + K] = color

    elif orientation == "horizontal":
        for start in range(0, N, K):
            color = 255 if ((start // K) % 2) == 0 else 0
            img[start : start + K, :] = color

    else:
        raise ValueError("orientation debe ser 'vertical' o 'horizontal'")

    return img

import numpy as np
import cv2


def generate_pattern_image(N, M, K, pattern="vertical"):
    """
    Genera una imagen de N×M píxeles con:
      - 'vertical': franjas verticales de ancho K
      - 'horizontal': franjas horizontales de alto K
      - 'squares': cuadrados alternos de tamaño K×K (patrón ajedrez)

    Parámetros
    ----------
    N : int
        Altura de la imagen en píxeles.
    M : int
        Anchura de la imagen en píxeles.
    K : int
        Dimensión de la franja o del cuadrado en píxeles.
    pattern : {'vertical', 'horizontal', 'squares'}
        Tipo de patrón a generar.

    Devuelve
    -------
    img : np.ndarray
        Imagen en escala de grises (dtype uint8) con valores 0 o 255.
    """
    img = np.zeros((N, M), dtype=np.uint8)

    if pattern == "vertical":
        for start in range(0, M, K):
            color = 255 if ((start // K) % 2) == 0 else 0
            img[:, start : start + K] = color

    elif pattern == "horizontal":
        for start in range(0, N, K):
            color = 255 if ((start // K) % 2) == 0 else 0
            img[start : start + K, :] = color

    elif pattern == "squares":
        for i in range(0, N, K):
            for j in range(0, M, K):
                # alterna color en función de la suma de índices de bloque
                color = 255 if (((i // K) + (j // K)) % 2) == 0 else 0
                img[i : i + K, j : j + K] = color

    else:
        raise ValueError("pattern debe ser 'vertical', 'horizontal' o 'squares'")

    return img


if __name__ == "__main__":
    # Parámetros de ejemplo
    altura = 400  # N
    anchura = 600  # M
    K = 50  # tamaño de banda o cuadrado

    # Generar y guardar cada patrón

        img = generate_pattern_image(altura, anchura, K, pattern=pat)
        filename = f"pattern_{pat}.png"
        cv2.imwrite(filename, img)
        print(f"Guardado {filename}")

C_k = np.fft.fft(x_n) / N  # Normalized Discrete Fourier Transform

# Reconstruct the signal using DTFS
x_reconstructed = np.zeros(N, dtype=complex)
for k in range(N):
    x_reconstructed += C_k[k] * np.exp(1j * 2 * np.pi * k * n / N)

plt.ylabel("Amplitude")
plt.title("DTFS: Original vs Reconstructed Signal")
plt.legend()
plt.grid()
plt.show()

import numpy as np
import matplotlib.pyplot as plt

# Parámetros
fs_cont = 5000  # Frecuencia de muestreo "continua" (muy alta)
fs_sampled = 1000  # Frecuencia de muestreo baja
T = 1  # Duración de la señal en segundos
f_signal = 30  # Frecuencia de la señal

# Señal continua
t_cont = np.linspace(0, T, fs_cont * T, endpoint=False)
signal_cont = np.sin(2 * np.pi * f_signal * t_cont)

# Señal muestreada
t_sampled = np.linspace(0, T, fs_sampled * T, endpoint=False)
signal_sampled = np.sin(2 * np.pi * f_signal * t_sampled)

# FFT continua
fft_cont = np.fft.fft(signal_cont)
freqs_cont = np.fft.fftfreq(len(signal_cont), d=1 / fs_cont)

# FFT muestreada
fft_sampled = np.fft.fft(signal_sampled)
freqs_sampled = np.fft.fftfreq(len(signal_sampled), d=1 / fs_sampled)

# Graficar espectro original y espectro muestreado
plt.figure(figsize=(12, 6))

plt.subplot(3, 1, 1)
plt.plot(
    t_cont, signal_cont
)
plt.subplot(3, 1, 2)
plt.plot(
    freqs_cont[: len(freqs_cont) // 2],
    np.abs(fft_cont[: len(freqs_cont) // 2]),
    label="Espectro original",
)
plt.title("Espectro de la señal continua")
plt.xlabel("Frecuencia (Hz)")
plt.ylabel("Magnitud")
plt.legend()

plt.subplot(3, 1, 3)
plt.plot(
    freqs_sampled[: len(freqs_sampled) // 2],
    np.abs(fft_sampled[: len(freqs_sampled) // 2]),
    label="Espectro muestreado",
)
plt.title("Espectro después del muestreo (repeticiones cada f_s)")
plt.xlabel("Frecuencia (Hz)")
plt.ylabel("Magnitud")
plt.legend()

plt.tight_layout()
plt.show()
import numpy as np
import matplotlib.pyplot as plt
from scipy.signal import tf2zpk

# Define numerator (zeros) and denominator (poles) coefficients of the transfer function
# Example: H(z) = (1 - 0.5z^-1) / (1 - 0.9z^-1)
b = [1, -0.5]  # Numerator coefficients (zeros)
a = [1, -0.9]  # Denominator coefficients (poles)

# Get zeros, poles, and gain
z, p, k = tf2zpk(b, a)

# Plot settings
fig, ax = plt.subplots()
ax.set_title("Pole-Zero Plot in the Z-Plane")

# Draw unit circle
unit_circle = plt.Circle((0, 0), 1, color="black", fill=False, linestyle="dashed")
ax.add_artist(unit_circle)

# Plot zeros and poles
ax.plot(np.real(z), np.imag(z), "go", label="Zeros")  # green circles
ax.plot(np.real(p), np.imag(p), "rx", label="Poles")  # red Xs

# Axes and formatting
ax.set_xlabel("Re")
ax.set_ylabel("Im")
ax.axhline(0, color="gray", linewidth=0.5)
ax.axvline(0, color="gray", linewidth=0.5)
ax.set_aspect("equal")
ax.grid(True, linestyle="--", alpha=0.5)
ax.legend()
ax.set_xlim(-1.5, 1.5)
ax.set_ylim(-1.5, 1.5)

plt.show()
def zpk_to_latex(z, p, k):
    def format_factor(value):
        sign = "+" if np.real(value) < 0 else "-"
        return f"(z {sign} {abs(np.real(value)):.2f})"

    num = " ".join([format_factor(zero) for zero in z]) if len(z) > 0 else "1"
    den = " ".join([format_factor(pole) for pole in p]) if len(p) > 0 else "1"

    latex_str = r"H(z) = " + f"{k:.2f} \\cdot \\frac{{{num}}}{{{den}}}"
    return latex_str
